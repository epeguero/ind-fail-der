\textsc{Lemma (Proof By Contrapositive)}: $\Gamma, \lnot B \vdash \lnot A \implies \Gamma, A \vdash B$

\begin{prooftree}

	\AxiomC{}	
	\RightLabel{\scriptsize L$\supset$}
	\UnaryInfC{$\Gamma, A \vdash B$}
	
					\AxiomC{}
					\RightLabel{\scriptsize id}					
					\UnaryInfC{$\Gamma, A, \bot \vdash B$}
			
			\RightLabel{\scriptsize L$\supset$}
			\BinaryInfC{$\Gamma, A, \lnot B \vdash B$}
			
							\AxiomC{}
							\RightLabel{\scriptsize id}
							\UnaryInfC{$\Gamma, A, B \vdash B$}							
							
					\RightLabel{\scriptsize EM}
					\BinaryInfC{$\Gamma, A \vdash B$}
										
\end{prooftree}
